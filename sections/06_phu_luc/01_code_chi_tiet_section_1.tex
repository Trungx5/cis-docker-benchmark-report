\subsection{Code chi tiết cho Section 1}\label{phu-luc-code-section-1}

\noindent\textbf{Chi tiết quy trình kiểm toán thủ công cho Section 1}

\begin{enumerate}
    \item \textbf{1.1.1 -- Tạo phân vùng riêng cho các container:}
    Xác định Docker root directory và kiểm tra xem có được mount riêng hay không:
{\footnotesize
\begin{verbatim}
# Bước 1: Xác định thư mục root của Docker
docker info -f '{{.DockerRootDir}}'
# Ví dụ output: /var/lib/docker

# Bước 2: Kiểm tra xem thư mục này có phải mount point riêng
mountpoint -q -- /var/lib/docker
echo $?
# Nếu in ra 0: PASS. Nếu in ra 1: FAIL.
\end{verbatim}
}

    \item \textbf{1.1.2 -- Chỉ cho phép các người dùng tin cậy kiểm soát Docker Daemon:}
    Xem danh sách các user nằm trong nhóm \texttt{docker} để thẩm định thủ công:
{\footnotesize
\begin{verbatim}
getent group docker
\end{verbatim}
}

    \item \textbf{1.1.3 -- Thiết lập kiểm toán (Auditing) cho Docker daemon:}
    Kiểm tra dịch vụ \texttt{auditd} và rule của dockerd:
{\footnotesize
\begin{verbatim}
# Kiểm tra auditd đã được cài đặt
apt list --installed | grep -w auditd

# Kiểm tra audit rule cho dockerd binary
auditctl -l | grep /usr/bin/dockerd
# Output mong đợi: -w /usr/bin/dockerd -k docker
\end{verbatim}
}

    \item \textbf{1.1.4 đến 1.1.18 -- Thiết lập kiểm toán cho các tệp tin và thư mục Docker cấu hình:}
    Chạy các lệnh tương ứng để rà soát sự hiện diện của rules trong auditd:
{\footnotesize
\begin{verbatim}
# 1.1.4 - Kiểm toán thư mục /run/containerd
auditctl -l | grep /run/containerd

# 1.1.5 - Kiểm toán thư mục /var/lib/docker
auditctl -l | grep /var/lib/docker

# 1.1.6 - Kiểm toán thư mục /etc/docker
auditctl -l | grep /etc/docker

# 1.1.7 - Kiểm toán tệp tin service của docker
SERVICE_PATH=$(systemctl show -p FragmentPath docker.service \
  | cut -d= -f2)
auditctl -l | grep "$SERVICE_PATH"

# 1.1.8 - Kiểm toán tệp tin containerd.sock socket
auditctl -l | grep /run/containerd/containerd.sock

# 1.1.9 - Kiểm toán tệp tin docker.socket socket
SOCKET_PATH=$(systemctl show -p FragmentPath docker.socket \
  | cut -d= -f2)
# Đọc ListenStream trong socket unit file để lấy path thực (ví dụ /run/docker.sock)
grep ListenStream "$SOCKET_PATH"
auditctl -l | grep /run/docker.sock

# 1.1.10 - Kiểm toán tệp tin /etc/default/docker
auditctl -l | grep /etc/default/docker

# 1.1.11 - Kiểm toán tệp tin /etc/docker/daemon.json
auditctl -l | grep /etc/docker/daemon.json

# 1.1.12 - Kiểm toán tệp tin /etc/containerd/config.toml
auditctl -l | grep /etc/containerd/config.toml

# 1.1.13 - Kiểm toán tệp tin /etc/sysconfig/docker (nếu có trên RHEL/CentOS)
auditctl -l | grep /etc/sysconfig/docker

# 1.1.14 - Kiểm toán tệp tin containerd binary
auditctl -l | grep /usr/bin/containerd

# 1.1.15 - Kiểm toán tệp tin containerd-shim binary (legacy)
auditctl -l | grep /usr/bin/containerd-shim

# 1.1.16 - Kiểm toán tệp tin containerd-shim-runc-v1 binary
auditctl -l | grep /usr/bin/containerd-shim-runc-v1

# 1.1.17 - Kiểm toán tệp tin containerd-shim-runc-v2 binary
auditctl -l | grep /usr/bin/containerd-shim-runc-v2

# 1.1.18 - Kiểm toán tệp tin runc binary
auditctl -l | grep /usr/bin/runc
\end{verbatim}
}

    \item \textbf{1.2.1 -- Đảm bảo máy chủ (Host) đã được làm cứng (Hardened):}
    Chạy công cụ đánh giá bảo mật hệ thống độc lập (Lynis):
{\footnotesize
\begin{verbatim}
sudo lynis audit system
# Kiểm tra chỉ số Hardening Index trong log
grep "Hardening index" /var/log/lynis-report.dat
\end{verbatim}
}

    \item \textbf{1.2.2 -- Đảm bảo phiên bản Docker được cập nhật mới nhất:}
    Kiểm tra phiên bản hiện hành và các bản cập nhật đang chờ:
{\footnotesize
\begin{verbatim}
docker version --format '{{.Server.Version}}'
sudo apt update
apt list --upgradeable | grep -E '^(docker-ce|docker-ce-cli|containerd.io)/'
\end{verbatim}
}
\end{enumerate}

\noindent\textbf{Chi tiết quy trình khắc phục thủ công cho Section 1}

\begin{enumerate}
    \item \textbf{1.1.1 -- Tạo phân vùng Docker riêng:}
    Nếu thư mục gốc của Docker (\texttt{/var/lib/docker}) nằm chung phân vùng hệ thống, hãy di chuyển nó sang một phân vùng riêng (ví dụ dùng LVM):
{\footnotesize
\begin{verbatim}
# Bước 1: Tạo logical volume mới và định dạng ext4
sudo lvcreate -L 50G -n docker vg_data
sudo mkfs.ext4 /dev/vg_data/docker

# Bước 2: Dừng dịch vụ Docker để tránh ghi đè dữ liệu
sudo systemctl stop docker

# Bước 3: Sao lưu dữ liệu Docker hiện tại sang thư mục tạm thời
sudo cp -rp /var/lib/docker /var/lib/docker.bak

# Bước 4: Mount phân vùng mới vào thư mục gốc của Docker
sudo mount /dev/vg_data/docker /var/lib/docker

# Bước 5: Cấu hình mount tự động khi khởi động lại hệ thống trong /etc/fstab
echo '/dev/vg_data/docker /var/lib/docker ext4 defaults 0 2' | sudo tee -a /etc/fstab

# Bước 6: Khởi chạy lại Docker dịch vụ
sudo systemctl start docker
\end{verbatim}
}

    \item \textbf{1.1.2 -- Chỉ cho phép người dùng tin cậy kiểm soát Docker daemon:}
    Gỡ bỏ những tài khoản không cần thiết khỏi nhóm \texttt{docker}:
{\footnotesize
\begin{verbatim}
sudo gpasswd -d <untrusted_user> docker
\end{verbatim}
}

    \item \textbf{1.1.3 đến 1.1.18 -- Khắc phục cấu hình kiểm toán (Auditing Rules):}
    Cài đặt dịch vụ \texttt{auditd} và ghi đè các cấu hình rules chuẩn bảo mật của Docker daemon và các tệp cấu hình:
{\footnotesize
\begin{verbatim}
# Bước 1: Cài đặt gói auditd
sudo apt install -y auditd

# Bước 2: Ghi danh sách rules bảo mật vào tệp cấu hình quy tắc của auditd
sudo tee /etc/audit/rules.d/docker.rules << 'EOF'
-w /usr/bin/dockerd -k docker
-a exit,always -F path=/run/containerd -F perm=war -k docker
-a exit,always -F path=/var/lib/docker -F perm=war -k docker
-w /etc/docker -k docker
-w /usr/lib/systemd/system/docker.service -k docker
-w /run/containerd/containerd.sock -k docker
-w /lib/systemd/system/docker.socket -k docker
-w /etc/default/docker -k docker
-w /etc/docker/daemon.json -k docker
-w /etc/containerd/config.toml -k docker
-w /usr/bin/containerd-shim-runc-v2 -k docker
-w /usr/bin/runc -k docker
-w /run/docker.sock -k docker
EOF

# Bước 3: Tải lại cấu hình quy tắc của auditd
sudo augenrules --load
\end{verbatim}
}

    \item \textbf{1.2.1 -- Làm cứng (Hardening) hệ điều hành Host:}
    Thực hiện các biện pháp làm cứng hệ thống cơ bản dựa trên báo cáo của Lynis:
{\footnotesize
\begin{verbatim}
# 1. Kích hoạt và cấu hình tường lửa cơ bản (UFW)
sudo ufw default deny incoming
sudo ufw default allow outgoing
sudo ufw allow 22/tcp # Đảm bảo không khóa SSH
sudo ufw enable

# 2. Vô hiệu hóa các dịch vụ mạng không cần thiết
sudo systemctl disable --now avahi-daemon
sudo systemctl disable --now cups

# 3. Làm cứng dịch vụ SSH (tắt root login trực tiếp)
sudo sed -i 's/PermitRootLogin yes/PermitRootLogin no/' /etc/ssh/sshd_config
sudo systemctl restart sshd
\end{verbatim}
}

    \item \textbf{1.2.2 -- Cập nhật phiên bản Docker lên bản mới nhất:}
    Nâng cấp các gói phần mềm liên quan đến Docker Engine:
{\footnotesize
\begin{verbatim}
sudo apt update
sudo apt upgrade -y docker-ce docker-ce-cli containerd.io \
  docker-buildx-plugin docker-compose-plugin
\end{verbatim}
}
\end{enumerate}

\noindent\textbf{Chi tiết kịch bản kiểm toán tự động cho Section 1}

Kịch bản kiểm toán tự động các khuyến nghị từ 1.1.1 đến 1.1.18 và từ 1.2.1 đến 1.2.2:

{\footnotesize
\begin{verbatim}
# Kiểm toán 1.1.1: Đảm bảo phân vùng riêng cho container
- name: 1.1.1 Ensure a separate partition for containers has been created
  ansible.builtin.command:
    cmd: "docker info -f '{{ .DockerRootDir }}'"
  register: docker_info

- name: Check user namespace
  ansible.builtin.shell:
    cmd: "docker info | grep -q userns"
  failed_when: false
  register: userns_output

- name: Set path to check (userns enabled)
  ansible.builtin.set_fact:
    docker_root_path: "{{ docker_info.stdout }}/.."
  when: userns_output.rc == 0

- name: Set path to check (userns disabled)
  ansible.builtin.set_fact:
    docker_root_path: "{{ docker_info.stdout }}"
  when: userns_output.rc == 1

- name: Check mount point
  ansible.builtin.command:
    cmd: 'mountpoint -q -- "{{ docker_root_path }}"'
  register: mount_point
  failed_when: false

- name: Pass / Fail Check 1.1.1
  ansible.builtin.assert:
    that:
      - mount_point.rc == 0
    success_msg: "[PASS] 1.1.1 - Separate mount point created"
    fail_msg: "[FAIL] 1.1.1 - No separate mount point created"
  register: assert_1_1_1
  failed_when: false

# Kiểm toán 1.1.2: Trusted users kiểm soát Docker Daemon
